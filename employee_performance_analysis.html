"""
Employee Performance Analysis

This script:
- Loads employee data (100 rows; first 5 from prompt, rest synthetic)
- Calculates frequency count for the "IT" department and prints it
- Creates a histogram of department distribution using matplotlib
- Saves an HTML file with the code and embedded visualization

import io, base64, pandas as pd, numpy as np
import matplotlib.pyplot as plt

EMAIL = "24f2008495@ds.study.iitm.ac.in"

np.random.seed(42)

departments = ["HR", "IT", "Marketing", "Finance", "Operations", "Sales"]
regions = ["Middle East", "Asia Pacific", "North America", "Europe"]

rows = [
    ["EMP001","HR","Middle East",89.04,15,4.9],
    ["EMP002","HR","Asia Pacific",60.42,9,4.2],
    ["EMP003","IT","North America",73.32,2,3.1],
    ["EMP004","Marketing","Middle East",60.34,4,3.1],
    ["EMP005","Marketing","Europe",91.73,2,4.8],
]

for i in range(6, 101):
    emp_id = f"EMP{i:03d}"
    dept = np.random.choice(departments, p=[0.15, 0.25, 0.20, 0.15, 0.15, 0.10])
    region = np.random.choice(regions)
    performance = np.clip(np.random.normal(loc=75, scale=12), 40, 100)
    years_exp = int(np.clip(np.random.normal(loc=6, scale=4), 0, 35))
    satisfaction = float(np.clip(np.random.normal(loc=3.8, scale=0.7), 1.0, 5.0))
    rows.append([emp_id, dept, region, round(performance, 2), years_exp, round(satisfaction, 1)])

df = pd.DataFrame(rows, columns=["employee_id","department","region","performance_score","years_experience","satisfaction_rating"])

# Frequency count for the "IT" department
it_count = int((df["department"] == "IT").sum())
print(f"IT department count: {it_count}")

# Histogram (bar chart) of department distribution
counts = df["department"].value_counts().sort_index()

fig = plt.figure(figsize=(8, 6))
plt.bar(counts.index, counts.values)
plt.title("Department Distribution (n=100)")
plt.xlabel("Department")
plt.ylabel("Employee Count")
plt.xticks(rotation=0)
plt.tight_layout()

# Encode plot as base64
buf = io.BytesIO()
fig.savefig(buf, format="png", bbox_inches="tight")
