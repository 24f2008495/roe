Code Sample — Simple CAGR

def cagr(begin, end, years):
    """Compound Annual Growth Rate"""
    return (end / begin) ** (1/years) - 1

# Example:
begin, end, years = 100, 141.0, 3
rate = cagr(begin, end, years)
print(f"CAGR: {rate:.2%}")
